       ```
       namespace App\Http\Middleware;
       use Closure;
       use Illuminate\Http\Request;
       use Symfony\Component\HttpFoundation\Response;
       
       class CacheHeaders {
           public function handle(Request $request, Closure $next, string $profile = 'vitrine'): Response {
               $response = $next($request);
               if (! $request->isMethod('GET')) return $response;
               $rules = [
                   'vitrine' => 'public, max-age=300, must-revalidate',
                   'sitemap' => 'public, max-age=3600, must-revalidate',
                   'health' => 'no-cache, no-store, must-revalidate',
               ];
               $value = $rules[$profile] ?? $rules['vitrine'];
               $response->headers->set('Cache-Control', $value);
               if ($profile === 'vitrine' || $profile === 'sitemap') {
                   $response->headers->set('Vary', 'Accept-Encoding');
               }
               return $response;
           }
       }
       ```
