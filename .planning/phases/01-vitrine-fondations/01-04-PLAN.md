<!-- Livewire ContactForm contract per RESEARCH Pattern 3 -->
public string $name = '';      Validate: required|string|max:80
public string $email = '';     Validate: required|email|max:160
public string $phone = '';     Validate: nullable|string|max:30
public string $message = '';   Validate: required|string|min:10|max:2000
public bool $sent = false;
public HoneypotData $extraFields;

submit() flow:
1. $this->rateLimit(5, 60) — danharrin trait throws TooManyRequestsException if exceeded
2. catch TooManyRequestsException → addError('throttle', 'Trop d\\'essais. Attendez quelques minutes puis réessayez.'); return
3. $this->protectAgainstSpam() — Spatie trait throws SpamException if honeypot tripped (silent rejection from user POV — set $sent=true to fake success)
4. $this->validate() — Livewire runs all #[Validate] attributes
5. Mail::to(config('contact.recipient'))->send(new ContactMessage(name, email, phone, message))
6. $this->sent = true; $this->reset(['name', 'email', 'phone', 'message'])

- Success state: replace form with sand-50 confirmation card containing H3 "Message envoyé." + "Pierre vous répondra rapidement. En attendant, écrivez-lui directement sur WhatsApp."
- Error states: inline text-danger text-sm below each invalid field; ring-danger on the field
